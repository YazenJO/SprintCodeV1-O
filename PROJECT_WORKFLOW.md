- ğŸ“„ **Document Analysis**: Upload PDFs, DOCX, TXT files

### OOP Principles Applied

**1. Encapsulation**
```typescript
// Question class hides internal state
class Question {
  private id: string;         // Hidden
  private text: string;       // Hidden
  
  public getText(): string {  // Controlled access
    return this.text;
  }
}
```

**2. Abstraction**
```typescript
// AIService defines contract, hides implementation
abstract class AIService {
  abstract generateQuestions(...): Promise<Question[]>;
}
```

**3. Inheritance**
```typescript
// GeminiAIService inherits from AIService
class GeminiAIService extends AIService {
  async generateQuestions(...): Promise<Question[]> {
    // Gemini-specific implementation
  }
}
```

**4. Polymorphism**
```typescript
// ExamService works with any AIService implementation
class ExamService {
  constructor(private aiService: AIService) {} // Could be Gemini, OpenAI, etc.
}
```

**5. Composition**
```typescript
// Exam HAS-A collection of Questions
class Exam {
  private questions: Question[];  // Composition over inheritance
}
```

**6. Dependency Injection**
```typescript
// Service is injected, not hardcoded
const geminiService = new GeminiAIService(apiKey);
const examService = new ExamService(geminiService);
```

```typescript
parseQuestionsFromResponse(response: string): Question[] {
  // 1. Extract JSON from markdown
  let jsonText = response;
  if (response.includes('```')) {
    const match = response.match(/```(?:json)?\s*(\[[\s\S]*?\])\s*```/);
    if (match) jsonText = match[1];
  }
  
  // 2. Clean up JSON
  jsonText = jsonText
    .replace(/^[^[]*/, '')  // Remove text before [
    .replace(/[^\]]*$/, '') // Remove text after ]
    .trim();
  
  // 3. Parse JSON
  const data = JSON.parse(jsonText);
  
  // 4. Validate structure
  if (!Array.isArray(data)) throw new Error('Invalid format');
  
  // 5. Create Question objects
  return data.map((item, index) => {
    return new Question(
      `q${index + 1}`,
      item.text,
      item.options,
      item.correctAnswer,
      item.explanation
    );
  });
}
```

const codePatterns = [

```typescript
function extractAndFormatCode(text: string) {
  // Extract code portion
  const match = text.match(
    /(?:code:|output of:?|following code:?|this code:?)\s*([\s\S]+?)(?:\?|$)/i
  );
  
  if (!match) return null;
  
  const code = match[1].trim();
  const questionText = text.replace(match[0], '').trim();
  
  // Auto-format code with line breaks
  const formattedCode = code
    .replace(/;\s*/g, ';\n')           // Semicolons â†’ newline
    .replace(/\{\s*/g, '{\n  ')        // Opening brace â†’ newline + indent
    .replace(/\}\s*/g, '\n}')          // Closing brace â†’ newline
    .replace(/\n\s*\n/g, '\n');        // Remove double newlines
  
  return { questionText, code: formattedCode };
}
```
